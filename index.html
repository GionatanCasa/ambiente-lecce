}).addTo(map);
 
// Caricamento dati in tempo reale
async function caricaPunti() {
const { data, error } = await supabaseClient
.from('segnalazioni')
.select('*');
 
if (error) console.error(error);
data.forEach(ponto => {
const marker = L.marker([ponto.lat, ponto.lng]).addTo(map);
marker.bindPopup(`
 
<b>Sopralluogo Ambiente</b><br>
Foglio: ${ponto.foglio} | Part: ${ponto.particella}<br>
Tecnico: ${ponto.tecnico}<br>

<img src="${ponto.foto_url}" class="popup-img">
</a>
`);
});
}
 
// Funzione Esporta Excel
async function esportaExcel() {
const { data } = await supabaseClient.from('segnalazioni').select('*');
const ws = XLSX.utils.json_to_sheet(data);
 
const wb = XLSX.utils.book_new();
XLSX.utils.book_append_sheet(wb, ws, "Sopralluoghi");
XLSX.writeFile(wb, "Report_Ambiente_Lecce.xlsx");
}
